92struct StereoColorSample
93{
95 StereoColorSample()
96 {
97 ts = 0;
98
99 // Default to black
100 r0 = g0 = b0 = 0;
101 r1 = g1 = b1 = 0;
102 }
103
105 StereoColorSample(unsigned long long time,
106 float red0, float green0, float blue0,
107 float red1, float green1, float blue1)
108 {
109 ts = time;
110 r0 = red0;
111 g0 = green0;
112 b0 = blue0;
113 r1 = red1;
114 g1 = green1;
115 b1 = blue1;
116 }
117
118 unsigned long long ts;
119 float r0;
120 float g0;
121 float b0;
122 float r1;
123 float g1;
124 float b1;
125
127 bool operator == (const StereoColorSample& a)
128 {
129 return isEqual(a);
130 }
131
133 bool operator != (const StereoColorSample& a)
134 {
135 return !isEqual(a);
136 }
137
139 inline bool isEqual(const StereoColorSample& a)
140 {
141 // Pointing to the same memory address?
142 if (&a == this)
143 return true;
144
145 // Otherewise compare values
146 return a.ts == ts &&
147 a.r0 == r0 && a.g0 == g0 && a.b0 == b0 &&
148 a.r1 == r1 && a.g1 == g1 && a.b1 == b1;
149 }
150
154 inline static StereoColorSample empty() { return { 0, 0, 0, 0, 0, 0, 0 }; }
155
157 inline static bool areEqual(const StereoColorSample& a, const StereoColorSample& b)
158 {
159 // Pointing to the same memory address?
160 if (&a == &b)
161 return true;
162
163 // Otherewise compare values
164 return a.ts == b.ts &&
165 a.r0 == b.r0 && a.g0 == b.g0 && a.b0 == b.b0 &&
166 a.r1 == b.r1 && a.g1 == b.g1 && a.b1 == b.b1;
167 }
168};
