53 // @brief Turns decibels into float gain factor
54 template <typename FloatType>
55 inline const FloatType dbToGain(const FloatType& dB)
56 {
57 return static_cast<FloatType>(std::exp(dB * (FloatType)M_20_LN10));
58 }
59
60 // @brief Turns float gain factor into decibels
61 template <typename FloatType>
62 inline const FloatType gainToDb(const FloatType& gain, FloatType infDb = -100)
63 {
64 if (gain < (FloatType)0.0000000298023223876953125)
65 return infDb;
66
67 const FloatType dB = std::log(gain) * (FloatType)M_LN10_20;
68 return (dB > infDb) ? dB : infDb;
69 }

78 template <typename FloatType>
79 inline bool deinterleaveStereoAudio(const FloatType* input, FloatType* outL, FloatType* outR, int samplesPerChannel)
80 {
81 if (input == nullptr || outL == nullptr || outR == nullptr || samplesPerChannel <= 0)
82 return false;
83
84 int sampleCount = 0;
85 int outChannels = 2; // stereo
86
87 for (int s = 0; s < samplesPerChannel; ++s)
88 {
89 outL[sampleCount] = input[s * outChannels];
90 outR[sampleCount++] = input[s * outChannels + 1];
91 }
92
93 return true;
94 }

105 template <typename FloatType>
106 inline bool deinterleaveQuadAudio(const FloatType* input, FloatType* out0, FloatType* out1, FloatType* out2, FloatType* out3, int samplesPerChannel)
107 {
108 if (input == nullptr || out0 == nullptr || out1 == nullptr || out2 == nullptr || out3 == nullptr || samplesPerChannel <= 0)
109 return false;
110
111 int sampleCount = 0;
112 int outChannels = 4; // quad channel audio
113
114 for (int s = 0; s < samplesPerChannel; ++s)
115 {
116 out0[sampleCount] = input[s * outChannels];
117 out1[sampleCount] = input[s * outChannels + 1];
118 out2[sampleCount] = input[s * outChannels + 2];
119 out3[sampleCount++] = input[s * outChannels + 3];
120 }
121
122 return true;
123 }

182 inline void setAmplitude(double newAmplitude) { amplitude = newAmplitude; }
183
185 inline double getAmplitude() const { return amplitude; }
186
190 void setFrequency(double newFrequencyHz)
191 {
192 frequency = newFrequencyHz;
193 phasePerSample = double(M_2PI) / (sampleRate / frequency);
194 }
195
197 inline double getFrequency() const { return frequency; }
198
202 void reset(double newSampleRate)
203 {
204 currentPhase = 0.0;
205 sampleRate = newSampleRate;
206 phasePerSample = double(M_2PI) / (sampleRate / frequency);
207 }
208
210 inline double nextSample()
211 {
212 double sample = amplitude * std::sin(currentPhase);
213 currentPhase += phasePerSample;
214 return sample;
215 }
216
220 inline void setPhase(double phase)
221 {
222 currentPhase = phase * double(M_2PI);
223 }
224
225 private:
226 double frequency;
227 double sampleRate;
228 double currentPhase;
229 double phasePerSample;
230 double amplitude;
231 };

295 class StaticLumasonicEncoder

301 StaticLumasonicEncoder(float maxLevelDb = LS_DEFAULT_REF_TONE_DB)
302 : cr0(0.f), cg0(0.f), cb0(0.f), cr1(0.f), cg1(0.f), cb1(0.f)
303 {
304 toneF.setAmplitude((double)dbToGain(maxLevelDb)); // reference to to full gain

306 toneG0.setAmplitude(0.); // color channel g to 0%
307 toneB0.setAmplitude(0.); // color channel b to 0%
308 toneR1.setAmplitude(0.); // color channel r to 0%
309 toneG1.setAmplitude(0.); // color channel g to 0%
310 toneB1.setAmplitude(0.); // color channel b to 0%
311
312 // Set each tone's unique frequency
313 toneF.setFrequency(LS_REF_TONE_FREQ); // reference tone
314 toneR0.setFrequency(LS_RED_TONE_FREQ); // left red
315 toneG0.setFrequency(LS_GREEN_TONE_FREQ); // left green
316 toneB0.setFrequency(LS_BLUE_TONE_FREQ); // left blue
317 toneR1.setFrequency(LS_RED_TONE_FREQ); // right red
318 toneG1.setFrequency(LS_GREEN_TONE_FREQ); // right green
319 toneB1.setFrequency(LS_BLUE_TONE_FREQ); // right blue
320 }

332 void setStereoColor(float r0, float g0, float b0, float r1, float g1, float b1)
333 {
334 cr0 = r0; cr1 = r1;
335 cg0 = g0; cg1 = g1;
336 cb0 = b0; cb1 = b1;
337 auto refLevel = dbToGain(LS_DEFAULT_REF_TONE_DB);
338 toneR0.setAmplitude((double)refLevel * (double)cr0);
339 toneG0.setAmplitude((double)refLevel * (double)cg0);
340 toneB0.setAmplitude((double)refLevel * (double)cb0);
341 toneR1.setAmplitude((double)refLevel * (double)cr1);
342 toneG1.setAmplitude((double)refLevel * (double)cg1);
343 toneB1.setAmplitude((double)refLevel * (double)cb1);
344 }

349 void reset(float sampleRate)
350 {
351 toneF.reset(sampleRate);
352 toneR0.reset(sampleRate); toneR1.reset(sampleRate);
353 toneG0.reset(sampleRate); toneG1.reset(sampleRate);
354 toneB0.reset(sampleRate); toneB1.reset(sampleRate);

363 void processBlock(float* in0, float* in1, int numSamples, bool replace = true)
364 {
365 if (replace)
366 {
367 for (int i = 0; i < numSamples; ++i)
368 {
369 auto f = toneF.nextSample();
370 auto r0 = toneR0.nextSample(); auto r1 = toneR1.nextSample();
371 auto g0 = toneG0.nextSample(); auto g1 = toneG1.nextSample();
372 auto b0 = toneB0.nextSample(); auto b1 = toneB1.nextSample();
373 in0[i] = (float)(f + r0 + g0 + b0);
374 in1[i] = (float)(f + r1 + g1 + b1);
375 }
376 }
377 else
378 {
379 for (int i = 0; i < numSamples; ++i)
380 {
381 auto f = toneF.nextSample();
382 auto r0 = toneR0.nextSample(); auto r1 = toneR1.nextSample();
383 auto g0 = toneG0.nextSample(); auto g1 = toneG1.nextSample();
384 auto b0 = toneB0.nextSample(); auto b1 = toneB1.nextSample();
385 in0[i] += (float)(f + r0 + g0 + b0);
386 in1[i] += (float)(f + r1 + g1 + b1);
387 }
388 }
389 }
390
391 private:
392 float cr0, cg0, cb0, cr1, cg1, cb1; // left & right red, green, and blue values
393 ToneGenerator toneF; // reference tone
394 ToneGenerator toneR0; // left red tone
395 ToneGenerator toneG0; // left green tone
396 ToneGenerator toneB0; // left blue tone
397 ToneGenerator toneR1; // right red tone
398 ToneGenerator toneG1; // right green tone
399 ToneGenerator toneB1; // right blue tone
400 };
